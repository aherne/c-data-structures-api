// Interface that defines operations one can perform on a GRAPH abstract data type
template<typename NODE, typename VALUE>
class Graph {
	// Clears all data allocated in heap memory
	virtual ~Graph(){}
	// Creates vertex and adds it to graph
	virtual NODE* createVertex(const VALUE& data) = 0;
	// Removes vertex from graph
	virtual void removeVertex(NODE* const& vertex) = 0;
	// Checks if there is a path of edges between vertexes
	virtual bool isPath(NODE* const& left, NODE* const& right) = 0;
	// Gets number of vertexes in graph
	virtual std::size_t getSize() const = 0;
	// Checks if graph contains a vertex by value
	virtual bool contains(const VALUE& data) const = 0;
	// Searches graph for a vertex by value
	virtual NODE* search(const VALUE& data) = 0;
};

// Abstract class that defines operations one can perform on an UNWEIGHTED GRAPH abstract data type where vertexes hold unique values on top of GRAPH abstract data type
template<typename T, int (*compare)(const T&, const T&) = comparator<T>, std::size_t (*hash)(const T&) = hash<T>>
class UnweightedGraph : public Graph<UnweightedGraphVertex<T, compare, hash>, T> {
	// Clears all data allocated in heap memory
	virtual ~UnweightedGraph()
	// Creates edge between vertexes
	virtual void createEdge(UnweightedGraphVertex<T, compare, hash>* const& left, UnweightedGraphVertex<T, compare, hash>* const& right) = 0
	// Removes edge between vertexes
	virtual void removeEdge(UnweightedGraphVertex<T, compare, hash>* const& left, UnweightedGraphVertex<T, compare, hash>* const& right) = 0
}

// Abstract class that defines operations one can perform on a WEIGHTED GRAPH abstract data type where vertexes hold unique values
template<typename T,typename W, int (*compare)(const T&, const T&) = comparator<T>, std::size_t (*hash)(const T&) = hash<T>>
class WeightedGraph : public Graph<WeightedGraphVertex<T,W,compare,hash>, T> {

	// Removes edge between vertexes

}
